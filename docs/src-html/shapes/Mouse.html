001package shapes;
002
003import java.awt.*;
004import java.awt.event.*;
005
006/**
007 * Captures input from the mouse. <code>Mouse</code>, along with
008 * {@link Keyboard}, is how you make your game interactive.
009 * <p>
010 * For example, to make a shape move towards the point where the mouse is
011 * clicked, you would write:
012 * <p>
013 * <code>shape.move(shape.towards(Mouse.clickLocation(), 10);</code>
014 */
015public class Mouse implements MouseMotionListener, MouseListener {
016  private static Point click;
017  
018  // will be null when mouse is not down
019  public static Point clickLocation() {
020    return click;
021  }
022
023  public void mousePressed(MouseEvent event) {
024    updateClick(event);
025  }
026
027  private void updateClick(MouseEvent event) {
028    click = new Point(
029      Geometry.fromCanvasX(event.getX()),
030      Geometry.fromCanvasY(event.getY())
031    );
032  }
033
034  public void mouseDragged(MouseEvent event) {
035    updateClick(event);
036  }
037
038  public void mouseReleased(MouseEvent event) {
039    click = null;
040  }
041
042  public void mouseMoved(MouseEvent event) {}
043  public void mouseExited(MouseEvent event) {}
044  public void mouseEntered(MouseEvent event) {}
045  public void mouseClicked(MouseEvent event) {}
046}
