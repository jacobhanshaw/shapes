001package shapes;
002
003import java.awt.Color;
004import java.awt.Graphics2D;
005
006public abstract class Shape {
007  private Color color;
008  private boolean fill; 
009  private boolean invisible; 
010  private boolean solid;
011  private String speech;
012  private int speechDuration;
013  private Color speechColor;
014  private boolean destroyed;
015  protected Direction direction;
016  private double speed;
017  protected Point center;
018
019  abstract public void setup();
020  abstract public void update();
021  abstract public void render(Graphics2D g);
022
023  // true if this shape contains s entirely
024  abstract public boolean contains(Shape s);
025  abstract public boolean contains(Point p);
026
027  public void autoUpdate() {
028    if (this.isSpeaking()) {
029      speechDuration--;
030    }
031    if (Math.abs(speed) > Geometry.EPSILON) {
032      move(getDirection(), speed);
033    }
034  }
035
036  // Overriding constructors should call super() and call setup() at the end
037  public Shape() {
038    Game.addShape(this);
039    // set default values
040    setColor(Color.GREEN);
041    setFill(true);
042    setSpeechColor(Color.BLACK);
043    destroyed = false;
044  }
045
046  public void setCenter(double x, double y) {
047    setCenter(new Point(x, y));
048  }
049
050  abstract public void move(Direction direction, double pixels);
051
052  public void move(double pixels) {
053    move(getDirection(), pixels);
054  }
055
056  boolean isTouching(Segment that) {
057    return Geometry.touching(this, that);
058  }
059
060  public boolean isTouching(Shape that) {
061    return Geometry.touching(this, that);
062  }
063
064  abstract public boolean isOffscreen();
065
066  public boolean isTouchingBorder() {
067    if (isOffscreen()) {
068      return true;
069    }
070    for (Segment border : Game.getCanvas().getBorders()) { 
071      if (isTouching(border)) {
072        return true;
073      }
074    }
075    return false;
076  }
077
078  public boolean isClicked() {
079    return this.contains(Mouse.clickLocation());
080  }
081
082  public void moveRight(double pixels) {
083    move(Direction.RIGHT, pixels);
084  }
085
086  public void moveLeft(double pixels) {
087    move(Direction.LEFT, pixels);
088  }
089
090  public void moveUp(double pixels) {
091    move(Direction.UP, pixels);
092  }
093  
094  public void moveDown(double pixels) {
095    move(Direction.DOWN, pixels);
096  }
097
098  public void say(String speech) {
099    this.speech = speech;
100    this.speechDuration = -1;
101  }
102  
103  public void say(String speech, int frames) {
104    this.speech = speech;
105    this.speechDuration = frames;
106  }
107
108  public String getSpeech() {
109    return speech;
110  }
111
112  public void setSpeechColor(Color speechColor) {
113    this.speechColor = speechColor;
114  }
115
116  public Color getSpeechColor() {
117    return speechColor;
118  }
119
120  public boolean isSpeaking() {
121    return speechDuration != 0;
122  }
123
124  // returns null if target is null
125  public Direction towards(Point target) {
126    if (target == null) return null;
127    Vector v = new Vector(this.getCenter(), target);
128    return v.getDirection();
129  }
130
131  public Direction towards(Shape target) {
132    return towards(target.getCenter());
133  }
134
135  public void destroy() {
136    destroyed = true;
137  }
138
139  public boolean isDestroyed() {
140    return destroyed;
141  }
142
143  public void setFill(boolean fill) {
144    this.fill = fill;
145  }
146
147  public boolean getFill() {
148    return fill;
149  }
150
151  public void setInvisible(boolean invisible) {
152    this.invisible = invisible;
153  }
154
155  public boolean getInvisible() {
156    return invisible;
157  }
158  
159  public void setColor(Color color) {
160    this.color = color;
161  }
162
163  public Color getColor() {
164    return color;
165  }
166
167  public void setSolid(boolean solid) {
168    if (this.solid == solid) {
169      return;
170    }
171
172    if (solid) {
173      Game.addSolid(this);
174    } else {
175      Game.removeSolid(this);
176    }
177
178    this.solid = solid;
179  }
180
181  public boolean getSolid() {
182    return solid;
183  }
184
185  public Direction getDirection() {
186    return direction;
187  }
188
189  public void setDirection(Direction direction) {
190    this.direction = direction;
191  }
192
193  public double getSpeed() {
194    return speed;
195  }
196
197  public void setSpeed(double speed) {
198    this.speed = speed;
199  }
200
201  public Point getCenter() {
202    return center;
203  }
204  
205  public void setCenter(Point center) {
206    this.center = center;
207  }
208}
