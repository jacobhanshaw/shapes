054  /**
055   * Updates the shape automatically.
056   *
057   * This update includes moving the shape when it has a speed, etc.
058   */
059  void autoUpdate() {
060    if (this.isSpeaking()) {
061      speechDuration--;
062    }
063    if (Math.abs(speed) > Geometry.EPSILON) {
064      move(getDirection(), speed);
065    }
066  }

120  /**
121   * Checks if this shape is touching the border of the window.
122   *
123   * @return  true if any part of the shape is touching the border of the
124   *          window, or if the shape is entirely outside the window, and false
125   *          otherwise.
126   */
127  public boolean isTouchingBorder() {
128    if (isOffscreen()) {
129      return true;
130    }
131    for (Segment border : Game.getCanvas().getBorders()) { 
132      if (isTouching(border)) {
133        return true;
134      }
135    }
136    return false;
137  }

139  /**
140   * Finds the farthest this shape can go towards a target with an obstacle in
141   * the way.
142   *
143   * If this shape tries to move to <code>target</code>, then
144   * <code>obstacle</code> may get in its way. Finds the farthest this shape
145   * can move towards <code>target</code>, and returns the location where the
146   * shape's center would be if it advanced as far as possible.
147   *
148   * @param   target    the point towards which this shape is moving.
149   * @param   obstacle  another shape which may obstruct this shape's movement.
150   * @return            a point representing the farthest this shape's center
151   *                    can move towards <code>target</code>.
152   */
153  abstract Point maxMovement(Point target, Shape obstacle);
154
155  /**
156   * Moves in the shape's direction.
157   *
158   * Moves <code>pixels</code> pixels in the direction set using
159   * <code>setDirection()</code>. Won't move through solid shapes.
160   *
161   * @param   pixels  the distance to move.
162   * @see     #move(Direction, double)
163   * @see     #setDirection(Direction)
164   */
165  public void move(double pixels) {
166    move(getDirection(), pixels);
167  }
168
169  /**
170   * Moves in the given direction.
171   *
172   * Moves <code>pixels</code> pixels in the given direction.
173   * Won't move through solid shapes.
174   *
175   * @param   pixels    the distance to move.
176   * @param   direction the direction in which to move.
177   * @see     #move(double)
178   */
179  public void move(Direction direction, double pixels) {
180    if (direction == null || Math.abs(pixels) < Geometry.EPSILON) {
181      return;
182    }
183    Point end = getCenter().translation(new Vector(direction, pixels));
184    System.out.println(end.toString());
185    Point maxMovement = end;
186    Set<Shape> solids = Game.getSolids();
187    for (Shape solid : solids) {
188      Point blockedEnd = this.maxMovement(end, solid);
189      if (Geometry.distance(getCenter(), blockedEnd) < Geometry.distance(getCenter(), maxMovement)) {
190        maxMovement = blockedEnd;
191      }
192    }
193    setCenter(maxMovement);
194  }

352  /**
353   * Returns the direction of a given point relative to this shape.
354   *
355   * Can be used to {@link #move} this shape towards a point:
356   * <code>this.move(this.towards(targetPoint), 10.0);</code>
357   *
358   * @param   target  the point to aim towards.
359   * @return          the direction of <code>target</code> relative to this
360   *                  shape, or null if <code>target</code> is null.
361   * @see             #towards(Shape)
362   * @see             #move(Direction, double)
363   * @see             #setDirection(Direction)
364   */
365  public Direction towards(Point target) {
366    if (target == null) return null;
367    Vector v = new Vector(this.getCenter(), target);
368    return v.getDirection();
369  }
370
371  /**
372   * Returns the direction of a shape relative to this shape.
373   *
374   * Can be used to {@link #move} this shape towards another shape:
375   * <code>this.move(this.towards(targetShape), 10.0);</code>
376   *
377   * @param   target  the shape to aim towards.
378   * @return          the direction of <code>target</code>'s relative to this
379   *                  shape's center.
380   * @see             #towards(Point)
381   * @see             #move(Direction, double)
382   * @see             #setDirection(Direction)
383   */
384  public Direction towards(Shape target) {
385    return towards(target.getCenter());
386  }

432  public void setSolid(boolean solid) {
433    if (this.solid == solid) {
434      return;
435    }
436
437    if (solid) {
438      Game.addSolid(this);
439    } else {
440      Game.removeSolid(this);
441    }
442
443    this.solid = solid;
444  }
