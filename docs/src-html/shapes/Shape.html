007/**
008 * A shape that appears on screen and interacts with other shapes.
009 * <p>
010 * Shapes are the basic building blocks of games built with the Shapes
011 * framework. They can be of different colors and sizes, can move around the
012 * screen, can collide with one another, and can say things, to name a few.
013 * <p>
014 * You won't use this class directly; you'll use one of its subclasses:
015 * {@link Circle}, {@link Rectangle} or {@link Triangle}.
016 *
017 * @author Nate Sullivan
018 * @version v0
019 */
020public abstract class Shape {
021  private Color color;
022  private boolean fill; 
023  private boolean invisible; 
024  private boolean solid;
025  private String speech;
026  private int speechDuration;
027  private Color speechColor;
028  private boolean destroyed;
029  protected Direction direction;

031  protected Point center;

033  /**
034   * Initializes the Shape. When you subclass shape, you'll
035   * override this method to do things like set the shape's color, set its
036   * starting position, etc.

039  /**
040   * Updates the shape once per frame. When you subclass shape, you'll
041   * override this method to do things like move the shape, change its
042   * size, etc.

067  /**
068   * Updates the shape automatically.
069   *
070   * This method moves the shape when it has a speed, etc.
071   */
072  void autoUpdate() {
073    if (this.isSpeaking()) {
074      speechDuration--;
075    }
076    if (Math.abs(speed) > Geometry.EPSILON) {
077      move(getDirection(), speed);
078    }
079  }

135   * Checks if this shape is touching the border of the window.

142    if (isOffscreen()) {

144    }
145    for (Segment border : Game.getBorders()) {

148      }

153  /**
154   * Finds the farthest this shape can go towards a target with an obstacle in
155   * the way.
156   * <p>
157   * If this shape tries to move to <code>target</code>, then
158   * <code>obstacle</code> may get in its way. Finds the farthest this shape
159   * can move towards <code>target</code>, and returns the location where the
160   * shape's center would be if it advanced as far as possible.
161   *
162   * @param   target    the point towards which this shape is moving.
163   * @param   obstacle  another shape which may obstruct this shape's movement.
164   * @return            a point representing the farthest this shape's center
165   *                    can move towards <code>target</code>.
166   */
167  abstract Point maxMovement(Point target, Shape obstacle);
168
169  /**
170   * Moves in the shape's direction.
171   * <p>
172   * Moves <code>pixels</code> pixels in the direction set using
173   * {@link #setDirection(Direction)}. Won't move through solid shapes.
174   *
175   * @param   pixels  the distance to move.
176   * @see     #move(Direction, double)
177   * @see     #setDirection(Direction)
178   */
179  public void move(double pixels) {
180    move(getDirection(), pixels);
181  }
182
183  /**
184   * Moves in the given direction.
185   * <p>
186   * Moves <code>pixels</code> pixels in the given direction.
187   * Won't move through solid shapes.
188   *
189   * @param   pixels    the distance to move.
190   * @param   direction the direction in which to move.
191   * @see     #move(double)
192   */
193  public void move(Direction direction, double pixels) {
194    if (direction == null || Math.abs(pixels) < Geometry.EPSILON) {
195      return;
196    }
197    Point end = getCenter().translation(new Vector(direction, pixels));
198    Point maxMovement = end;
199    Set<Shape> solids = Game.getSolids();
200    for (Shape solid : solids) {
201      if (solid == this) continue;
202      Point blockedEnd = this.maxMovement(end, solid);
203      if (Geometry.distance(getCenter(), blockedEnd) < Geometry.distance(getCenter(), maxMovement)) {
204        maxMovement = blockedEnd;
205      }
206    }
207    setCenter(maxMovement);
208  }

210  /**
211   * Checks if this shape was clicked since the previous frame.
212   *
213   * Returns true if:
214   * <ul>
215   *  <li>the mouse button was clicked inside this shape since the last
216   *  frame</li>
217   *  <li>the mouse was dragged into this shape while the mouse button was
218   *  down, and is currently inside this shape</li>
219   * </ul>
220   *
221   * @return  true if this shape was clicked since the previous frame, false otherwise.
222   * @see     Mouse#clickLocation()
223   */
224  public boolean isClicked() {
225    if (Mouse.clickLocation() == null) {
226      return false;
227    }
228
229    return this.contains(Mouse.clickLocation());
230  }

363  /**
364   * Returns the direction of a given point relative to this shape.
365   * <p>
366   * Can be used to {@link #move} this shape towards a point:
367   * <code>this.move(this.towards(targetPoint), 10.0);</code>
368   *
369   * @param   target  the point to aim towards.
370   * @return          the direction of <code>target</code> relative to this
371   *                  shape, or null if <code>target</code> is null.
372   * @see             #towards(Shape)
373   * @see             #move(Direction, double)
374   * @see             #setDirection(Direction)
375   */
376  public Direction towards(Point target) {
377    if (target == null) return null;
378    Vector v = new Vector(this.getCenter(), target);
379    return v.getDirection();
380  }

382  /**
383   * Returns the direction of a shape relative to this shape.
384   * <p>
385   * Can be used to {@link #move} this shape towards another shape:
386   * <code>this.move(this.towards(targetShape), 10.0);</code>
387   *
388   * @param   target  the shape to aim towards.
389   * @return          the direction of <code>target</code>'s relative to this
390   *                  shape's center.
391   * @see             #towards(Point)
392   * @see             #move(Direction, double)
393   * @see             #setDirection(Direction)
394   */
395  public Direction towards(Shape target) {
396    return towards(target.getCenter());

511  /**
512   * Set whether other shapes can overlap with this shape.
513   * <p>
514   * An example of a solid shape is a wall that other shapes can't pass
515   * through.
516   * <p>
517   * No shape can overlap with a solid shape. A non-solid shape can overlap
518   * with other non-solid shapes.
519   *
520   * @param solid true for a shape that can't be overlapped, false for a shape
521   *              that can be.
522   */
523  public void setSolid(boolean solid) {
524    if (this.solid == solid) {
525      return;
526    }
527
528    if (solid) {
529      Game.addSolid(this);
530    } else {
531      Game.removeSolid(this);
532    }
533
534    this.solid = solid;
535  }

545   *
