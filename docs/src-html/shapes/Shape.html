007/**
008 * Represents a shape that is drawn to the screen and can interact with other
009 * shapes.
010 * <p>
011 * Shapes are the basic building blocks of games built with the Shapes
012 * framework. They can be of different colors and sizes, can move around the
013 * screen, can collide with one another, and can say things, to name a few.
014 * <p>
015 * You won't use this class directly; you'll use one of its subclasses:
016 * {@link Circle}, {@link Rectangle} or {@link Triangle}.
017 *
018 * @author Nate Sullivan
019 * @version v0
020 */
021public abstract class Shape {

068  /**
069   * Updates the shape automatically.
070   *
071   * This method moves the shape when it has a speed, etc.
072   */
073  void autoUpdate() {
074    if (this.isSpeaking()) {
075      speechDuration--;
076    }
077    if (Math.abs(speed) > Geometry.EPSILON) {
078      move(getDirection(), speed);
079    }
080  }

154  /**
155   * Finds the farthest this shape can go towards a target with an obstacle in
156   * the way.
157   * <p>
158   * If this shape tries to move to <code>target</code>, then
159   * <code>obstacle</code> may get in its way. Finds the farthest this shape
160   * can move towards <code>target</code>, and returns the location where the
161   * shape's center would be if it advanced as far as possible.
162   *
163   * @param   target    the point towards which this shape is moving.
164   * @param   obstacle  another shape which may obstruct this shape's movement.
165   * @return            a point representing the farthest this shape's center
166   *                    can move towards <code>target</code>.
167   */
168  abstract Point maxMovement(Point target, Shape obstacle);
169
170  /**
171   * Moves in the shape's direction.
172   * <p>
173   * Moves <code>pixels</code> pixels in the direction set using
174   * {@link #setDirection(Direction)}. Won't move through solid shapes.
175   *
176   * @param   pixels  the distance to move.
177   * @see     #move(Direction, double)
178   * @see     #setDirection(Direction)
179   */
180  public void move(double pixels) {
181    move(getDirection(), pixels);
182  }
183
184  /**
185   * Moves in the given direction.
186   * <p>
187   * Moves <code>pixels</code> pixels in the given direction.
188   * Won't move through solid shapes.
189   *
190   * @param   pixels    the distance to move.
191   * @param   direction the direction in which to move.
192   * @see     #move(double)
193   */
194  public void move(Direction direction, double pixels) {
195    if (direction == null || Math.abs(pixels) < Geometry.EPSILON) {
196      return;
197    }
198    Point end = getCenter().translation(new Vector(direction, pixels));
199    Point maxMovement = end;
200    Set<Shape> solids = Game.getSolids();
201    for (Shape solid : solids) {
202      if (solid == this) continue;
203      Point blockedEnd = this.maxMovement(end, solid);
204      if (Geometry.distance(getCenter(), blockedEnd) < Geometry.distance(getCenter(), maxMovement)) {
205        maxMovement = blockedEnd;
206      }
207    }
208    setCenter(maxMovement);
209  }

361  /**
362   * Returns the direction of a given point relative to this shape.
363   * <p>
364   * Can be used to {@link #move} this shape towards a point:
365   * <code>this.move(this.towards(targetPoint), 10.0);</code>
366   *
367   * @param   target  the point to aim towards.
368   * @return          the direction of <code>target</code> relative to this
369   *                  shape, or null if <code>target</code> is null.
370   * @see             #towards(Shape)
371   * @see             #move(Direction, double)
372   * @see             #setDirection(Direction)
373   */
374  public Direction towards(Point target) {
375    if (target == null) return null;
376    Vector v = new Vector(this.getCenter(), target);
377    return v.getDirection();
378  }

385   *
386   * @param   target  the shape to aim towards.
387   * @return          the direction of <code>target</code>'s relative to this
388   *                  shape's center.
389   * @see             #towards(Point)
390   * @see             #move(Direction, double)
391   * @see             #setDirection(Direction)
392   */
393  public Direction towards(Shape target) {

509  /**
510   * Set whether other shapes can overlap with this shape.
511   * <p>
512   * An example of a solid shape is a wall that other shapes can't pass
513   * through.
514   * <p>
515   * No shape can overlap with a solid shape. A non-solid shape can overlap
516   * with other non-solid shapes.
517   *
518   * @param solid true for a shape that can't be overlapped, false for a shape
519   *              that can be.
520   */
521  public void setSolid(boolean solid) {
522    if (this.solid == solid) {
523      return;
524    }
525
526    if (solid) {
527      Game.addSolid(this);
528    } else {
529      Game.removeSolid(this);
530    }
531
532    this.solid = solid;
533  }
