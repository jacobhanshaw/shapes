001package shapes;
002
003public class Direction {
004  double degrees;
005
006  public static Direction RIGHT = new Direction(0.0);
007  public static Direction LEFT = new Direction(180.0);
008  public static Direction DOWN = new Direction(270.0);
009  public static Direction UP = new Direction(90.0);
010
011  public Direction(double degrees) {
012    setDegrees(degrees);
013  }
014
015  public double getRadians() {
016    return degrees / 180.0 * Math.PI;
017  }
018
019  public void setRadians(double radians) {
020    double degrees = radians / Math.PI * 180.0;
021    setDegrees(degrees);
022  }
023
024  public double getDegrees() {
025    return degrees;
026  }
027
028  public void setDegrees(double degrees) {
029    degrees %= 360.0;
030    if (degrees < 0.0) {
031      degrees += 360.0;
032    }
033    this.degrees = degrees;
034  }
035
036  public Direction reverse() {
037    return new Direction(degrees + 180.0);
038  }
039
040  public Direction perpendicular() {
041    return new Direction(degrees + 90.0);
042  }
043
044  public static Direction random() {
045    return new Direction(Math.random() * 360.0);
046  }
047
048  public String toString() {
049    return degrees + " degrees";
050  }
051}
