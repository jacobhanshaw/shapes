001package shapes;
002
003import java.awt.event.KeyEvent;
004
005/**
006 * Used in calls to {@link Keyboard#direction(KeySet)} to specify which
007 * direction keys to listen to. Either the arrow keys or WASD.
008 */
009public enum KeySet {
010  ARROWS(KeyEvent.VK_UP, KeyEvent.VK_DOWN, KeyEvent.VK_LEFT, KeyEvent.VK_RIGHT),
011  WASD(KeyEvent.VK_W, KeyEvent.VK_S, KeyEvent.VK_A, KeyEvent.VK_D);
012
013  public int UP, DOWN, LEFT, RIGHT;
014
015  private KeySet(int up, int down, int left, int right) {
016    UP = up;
017    DOWN = down;
018    LEFT = left;
019    RIGHT = right;
020  }
021
022  public Vector getVector(KeyEvent key) {
023    int code = key.getKeyCode();
024    if (code == UP)
025      return Vector.UP;
026    if (code == DOWN)
027      return Vector.DOWN;
028    if (code == LEFT)
029      return Vector.LEFT;
030    if (code == RIGHT)
031      return Vector.RIGHT;
032
033    return null;
034  }
035}
