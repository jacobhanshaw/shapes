005public abstract class ConvexPolygon extends Shape {
006
007  abstract Point[] getUnrotatedCorners();
008
009  public Point[] getCorners() {
010    Point[] corners = getUnrotatedCorners();
011    if (direction == null || Math.abs(direction.getRadians()) < Geometry.EPSILON) {
012      return corners;
013    }
014    double angle = direction.getRadians();
015    for (int i = 0; i < corners.length; i++) {
016      double xOffset = corners[i].getX() - center.getX();
017      double yOffset = corners[i].getY() - center.getY();
018      corners[i] = new Point(
019        center.getX() + xOffset * Math.cos(angle) - yOffset * Math.sin(angle),
020        center.getY() + xOffset * Math.sin(angle) + yOffset * Math.cos(angle)
021      );
022    }
023
024    return corners;
025  }
026
027
028  public boolean isOffscreen() {
029    for (Point corner : getCorners()) {
030      if (!Geometry.offscreen(corner)) {
031        return false;
032      }
033    }
034    return true;
035  }

041  public boolean contains(Point point) {
042    boolean result = false;
043    Point[] corners = getCorners();
044    for (int i = 0, j = corners.length - 1; i < corners.length; j = i++) {
045      // icky math, copied from Stack Overflow 8721406
046      if (
047        (corners[i].getY() > point.getY()) != (corners[j].getY() > point.getY()) &&
048        (point.getX() < (corners[j].getX() - corners[i].getX()) * (point.getY() - corners[i].getY()) / (corners[j].getY()-corners[i].getY()) + corners[i].getX())
049      ) {
050        result = !result;
051      }
052    }
053    return result;
054  }
055
056  public boolean contains(Shape shape) {
057    if (shape instanceof ConvexPolygon) {
058      Point[] corners = ((ConvexPolygon) shape).getCorners();
059      for (Point corner : corners) {
060        if (!this.contains(corner)) {
061          return false;
062        }
063      }
064      return true;
065    } else if (shape instanceof Circle) {

067      return false;
068    }
069    return false;
070  }

076  public void render(Graphics2D g) {
077    renderSpeech(g);
078    if (this.getInvisible()) {
079      return;
080    }
081
082    g.setColor(getColor());
083    Point[] corners = getCorners();
084    int[] x = new int[corners.length];
085    int[] y = new int[corners.length];
086    for (int i = 0; i < corners.length; i++) {
087      x[i] = (int)corners[i].getCanvasX();
088      y[i] = (int)corners[i].getCanvasY();
089    }
090    if (this.getFill()) {
091      g.fillPolygon(x, y, corners.length);
092    } else {
093      g.drawPolygon(x, y, corners.length);
094    }
095  }
