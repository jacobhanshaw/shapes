007import java.awt.*;
008import java.awt.geom.*;
009import java.*;
010import javax.swing.*;
011import java.util.*;
012import java.util.concurrent.*;
013
014public abstract class Game {
015  private static JFrame frame;
016  private static Canvas canvas;
017  // TODO: figure out what the fuck Game is.
018  // static methods?
019  // singleton?
020  // passed as a parameter?
021  private static Set<Shape> solidShapes;
022  private static Set<Shape> allShapes;
023
024  public static final int HEIGHT = 500;
025  public static final int WIDTH = 500;
026  
027  public Game() {
028    canvas = new Canvas(this);
029    solidShapes =
030      Collections.newSetFromMap(new ConcurrentHashMap<Shape, Boolean>());
031    allShapes = 
032      Collections.newSetFromMap(new ConcurrentHashMap<Shape, Boolean>());
033
034    frame = new JFrame();
035    Mouse mouse = new Mouse();
036    frame.addMouseMotionListener(mouse);
037    frame.addMouseListener(mouse);
038    frame.addKeyListener(new Keyboard());
039    frame.setDefaultCloseOperation(frame.EXIT_ON_CLOSE);
040
041    setBackgroundColor(Color.BLUE);
042    setup();
043
044    frame.add(canvas);
045    frame.pack();
046    frame.setLocationRelativeTo(null);
047    frame.setResizable(false);
048    frame.setVisible(true);
049  }
050
051  // Override this!
052  public void setup() {
053  }
054
055  // Override this!
056  public void update() {
057  }
058
059  void autoUpdate() {
060    for (Shape s : allShapes) {
061      s.autoUpdate();
062    }
063
064    Iterator<Shape> iter = allShapes.iterator();
065    while (iter.hasNext()) {
066      Shape s = iter.next();
067      s.update();
068      if (s.isDestroyed()) {
069        iter.remove();
070      }
071    }
072  }
073
074  static Set<Shape> getSolids() {
075    return solidShapes;
076  }
077
078  static void addSolid(Shape shape) {
079    solidShapes.add(shape);
080  }
081
082  static void removeSolid(Shape shape) {
083    solidShapes.remove(shape);
084  }
085
086  static void addShape(Shape shape) {
087    allShapes.add(shape);
088  }
089
090  static void removeShape(Shape shape) {
091    allShapes.remove(shape);
092  }
093
094  public static Shape[] getAllShapes() {
095    return allShapes.toArray(new Shape[0]);
096  }
097
098  public static Color getBackgroundColor() {
099    return canvas.getBackground();
100  }
101
102  public static void setBackgroundColor(Color backgroundColor) {
103    canvas.setBackground(backgroundColor);
104  }
105
106  public static Point[] getCorners() {
107    return canvas.getCorners();
108  }
109
110  static Segment[] getBorders() {
111    return canvas.getBorders();
112  }
113}
