001package shapes;
002
003public class Rectangle extends ConvexPolygon {
004  private double width, height;
005
006  public Rectangle(Point center, double width, double height) {
007    setDisplaysRotation(true);
008    this.center = center;
009    this.width = width;
010    this.height = height;
011  }
012
013  public Rectangle() {
014    this(new Point(100, 100), 50, 100);
015  }
016
017  // Override this!
018  public void setup() {}
019
020  // Override this!
021  public void update() {}
022
023  Point[] getUnrotatedCorners() {
024    return new Point[] {
025      getCenter().translation(new Vector(-1 * width / 2, height / 2)),
026      getCenter().translation(new Vector(width / 2,      height / 2)),
027      getCenter().translation(new Vector(width / 2,      -1 * height / 2)),
028      getCenter().translation(new Vector(-1 * width / 2, -1 * height / 2))
029    };
030  }
031
032  public double getHeight() {
033    return height;
034  }
035
036  public void setHeight(double height) {
037    this.height = height;
038  }
039
040  public double getWidth() {
041    return width;
042  }
043
044  public void setWidth(double width) {
045    this.width = width;
046  }
047}
