001package shapes;
002
003import java.awt.event.*;
004import java.util.*;
005import java.util.concurrent.*;
006
007/**
008 * Captures input from the keyboard. <code>Keyboard</code>, along with
009 * {@link Mouse}, is how you make your game interactive.
010 * <p>
011 * For example, to make a shape move according to the arrow keys, you would
012 * write:
013 * <p>
014 * <code>shape.move(Keyboard.direction(), 10);</code>
015 */
016public class Keyboard extends KeyAdapter {
017  private static Set<KeyEvent> keysPressed;
018  private static KeyEvent mostRecentKeyPressed;
019
020  public Keyboard() {
021    if (keysPressed == null) {
022      keysPressed =
023        Collections.newSetFromMap(new ConcurrentHashMap<KeyEvent, Boolean>());
024    }
025  }
026
027  public static boolean keyIsPressed(String key) {
028    for (KeyEvent event : keysPressed) {
029      if (key.equals(KeyEvent.getKeyText(event.getKeyCode()))) {
030        return true;
031      }
032    }
033    return false;
034  }
035
036  public static String[] keys() {
037    Iterator<KeyEvent> keyEventIterator = keysPressed.iterator();
038    String[] keys = new String[keysPressed.size()];
039    for (int i = 0; i < keys.length && keyEventIterator.hasNext(); i++) {
040      keys[i] = KeyEvent.getKeyText(keyEventIterator.next().getKeyCode());
041    }
042
043    return keys;
044  }
045
046  public static String key() {
047    if (mostRecentKeyPressed == null) {
048      return null;
049    }
050
051    return KeyEvent.getKeyText(mostRecentKeyPressed.getKeyCode());
052  }
053
054  public static Direction direction() {
055    return direction(KeySet.ARROWS);
056  }
057
058  // returns null if no arrows are pressed, or if 3+ arrows are pressed, or if
059  // opposing keys are pressed
060  public static Direction direction(KeySet set) {
061    ArrayList<Vector> vectorsPressed = new ArrayList<Vector>();
062    for (KeyEvent keyPressed : keysPressed) {
063      Vector vectorPressed = set.getVector(keyPressed);
064      if (vectorPressed != null) {
065        vectorsPressed.add(vectorPressed);
066      }
067    }
068
069    if (vectorsPressed.size() == 0) {
070      return null;
071    }
072    Vector ret = vectorsPressed.get(0);
073    for (int i = 1; i < vectorsPressed.size(); i++) {
074      ret = ret.add(vectorsPressed.get(i));
075    }
076    if (Math.abs(ret.getMagnitude()) < Geometry.EPSILON) {
077      return null;
078    }
079    return ret.getDirection();
080  }
081
082  @Override
083  public void keyPressed(KeyEvent keyPressed) {
084    keysPressed.add(keyPressed);
085    mostRecentKeyPressed = keyPressed;
086  }
087
088  @Override
089  public void keyReleased(KeyEvent keyReleased) {
090    // remove from keysPressed
091    Iterator<KeyEvent> iter = keysPressed.iterator();
092    while (iter.hasNext()) {
093      if (iter.next().getKeyCode() == keyReleased.getKeyCode()) {
094        iter.remove();
095      }
096    }
097
098    if (mostRecentKeyPressed != null &&
099        mostRecentKeyPressed.getKeyCode() == keyReleased.getKeyCode()) {
100      mostRecentKeyPressed = null;
101    }
102  }
103}
